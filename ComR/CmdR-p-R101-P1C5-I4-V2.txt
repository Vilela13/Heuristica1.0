require(ggplot2) 
Dados <- data.frame(nomes = c("C1(1)","C2(3)","C3(5)","C4(3)","C5(2)","P1")

, y <- c(5,40,10,49,5,35)
, tipo <- c(3,3,3,3,3,4)
, tamanho <- c(1,1,1,1,1,2)
)
pdf('p-R101-P1C5-I4-V2.pdf')
