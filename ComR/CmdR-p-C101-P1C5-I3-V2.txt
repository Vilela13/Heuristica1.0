require(ggplot2) 
Dados <- data.frame(nomes = c("C1(4)","C2(1)","C3(5)","C4(1)","C5(4)","P1")
, x <- c(45,28,25,30,22,40)
, y <- c(70,52,85,50,85,50)
, tipo <- c(3,3,3,3,3,4)
, tamanho <- c(1,1,1,1,1,2)
)
pdf('p-C101-P1C5-I3-V2.pdf')
